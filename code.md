## Question one
```c
#include<stdio.h>
#include<math.h>
int main()
{
	float a,b,c;
	double x1 = 0,x2 = 0;
	scanf("%f %f %f",&a,&b,&c);
	if((b*b-4*a*c) >= 0)
	{
		x1=(-b+sqrt(b*b-4*a*c))/(2*a);
		x2=(-b-sqrt(b*b-4*a*c))/(2*a);
	}
	printf("x1=%.2lf,x2=%.2lf\n",x1,x2);
	return 0;
}
```

		i++;
		scanf("%d",&score[i]);

		for(k=0;k< 9 - j;k++)
			if(score[k]>score[k+1])
			{
                temp=score[k];
                score[k]=score[k+1];
			    score[k+1]=temp;
			}

#include <stdio.h>
int main()
{
	int N = 0;
	int a = 0,b = 0, c= 0;
	scanf("%d", &N);
	N /= 10;
	while(N != 0)
	{
		if(N >= 5)
		{
			a++;
			N -= 5;
		}
		else if(N >= 3)
		{
			b++;
			N -= 3;
		}
		else
		{
			c++;
			N -= 1;
		}
	}
	printf("%d\n", (7*a + 4*b + c));

}
